
on_actions = {
	on_daily = {
		effect = {
			if = {
				limit = {
					has_game_rule = {
						rule = naval_cap
						option = on
					}
				}
				NVLCAP_check_fleet_size = yes 
			}
		}
	}
	
	on_startup = {
		effect = {
			if = {
				limit = {
					has_game_rule = {
						rule = naval_cap
						option = on
					}
				}
				VEN = { NVLCAP_startup_on_actions = yes }
				
				# Variables to re-use elsewhere
				set_variable = { global.CAP = 60 }
				set_variable = { global.SSCAP = 1 }
				set_variable = { global.DDCAP = 0.2 }
				set_variable = { global.LCCAP = 1 }
				set_variable = { global.HCCAP = 1 }
				set_variable = { global.BCCAP = 2 }
				# Not used in Vanilla
				#set_variable = { global.SHBBCAP = 1 } 
				#set_variable = { global.SHCVCAP = 1 }
			}
		}
		
		effect = {
			if = {
				limit = {
					has_game_rule = {
						rule = naval_cap
						option = ONEFIFTYCAP
					}
				}
				# Set variable to 150
				set_variable={GLOBAL.cap = 150}
			}else_if = {
				limit = {
					has_game_rule = {
						rule = naval_cap
						option = ONEHUNDREDCAP
					}
				}
				set_variable={GLOBAL.cap = 100}
			}else = {
				set_variable={GLOBAL.cap = 0}
			}
		}
	}
}
